## Complete Code — MoreTray Component

```tsx
// client/src/components/MoreTray.tsx

import { useState } from 'react';
import { useTranslation } from 'react-i18next';
import { useNavigate } from 'react-router-dom';

interface MoreTrayProps {
  isOpen: boolean;
  onClose: () => void;
}

export function MoreTray({ isOpen, onClose }: MoreTrayProps) {
  const { t } = useTranslation();
  const navigate = useNavigate();

  const mainFeatures = [
    {
      icon: '🫁',
      label: t('menu_prana'),
      sublabel: t('more.prana_sub'),
      path: '/prana',
      color: '#4CAF50'
    },
    {
      icon: '📖',
      label: t('menu_review'),
      sublabel: t('more.review_sub'),
      path: '/review',
      color: '#FF9800'
    },
    {
      icon: '🎨',
      label: t('menu_customize'),
      sublabel: t('more.customize_sub'),
      path: null,
      action: 'customize',
      color: '#9C27B0'
    },
    {
      icon: '🧠',
      label: t('menu_memory'),
      sublabel: t('more.memory_sub'),
      path: '/memory',
      color: '#2196F3'
    },
    {
      icon: '👥',
      label: t('menu_community'),
      sublabel: t('more.community_sub'),
      path: '/community',
      color: '#FF5722'
    },
  ];

  const utilityLinks = [
    {
      icon: '🔒',
      label: t('menu_privacy'),
      path: '/privacy-control'
    },
    {
      icon: '📣',
      label: t('menu_report'),
      path: null,
      action: 'report'
    },
  ];

  function handleItem(item: any) {
    onClose();
    if (item.path) {
      navigate(item.path);
    } else if (item.action === 'customize') {
      // Open customize panel
      window.dispatchEvent(new CustomEvent('open-customize'));
    } else if (item.action === 'report') {
      window.dispatchEvent(new CustomEvent('open-report'));
    }
  }

  if (!isOpen) return null;

  return (
    <>
      {/* Backdrop */}
      <div
        className="more-tray-backdrop"
        onClick={onClose}
      />

      {/* Tray */}
      <div className={`more-tray ${isOpen ? 'open' : ''}`}>

        {/* Drag handle */}
        <div className="more-tray-handle" />

        {/* Header */}
        <div className="more-tray-header">
          <span className="more-tray-title">
            {t('more.title')}
          </span>
        </div>

        {/* Main features */}
        <div className="more-tray-features">
          {mainFeatures.map(item => (
            <button
              key={item.label}
              className="more-tray-item"
              onClick={() => handleItem(item)}
            >
              <div
                className="mti-icon"
                style={{ background: `${item.color}18`,
                         border: `1px solid ${item.color}30` }}
              >
                {item.icon}
              </div>
              <div className="mti-text">
                <span className="mti-label">{item.label}</span>
                <span className="mti-sub">{item.sublabel}</span>
              </div>
              <span className="mti-arrow">›</span>
            </button>
          ))}
        </div>

        {/* Divider */}
        <div className="more-tray-divider" />

        {/* Utility links */}
        <div className="more-tray-utility">
          {utilityLinks.map(item => (
            <button
              key={item.label}
              className="more-tray-util-item"
              onClick={() => handleItem(item)}
            >
              <span className="mtu-icon">{item.icon}</span>
              <span className="mtu-label">{item.label}</span>
            </button>
          ))}
        </div>

      </div>
    </>
  );
}
```

## Updated Bottom Nav

```tsx
// client/src/components/BottomNav.tsx

export function BottomNav() {
  const { t } = useTranslation();
  const [moreTrayOpen, setMoreTrayOpen] = useState(false);
  const location = useLocation();

  const navItems = [
    { icon: '🏠', label: t('nav.home'),  path: '/' },
    { icon: '🎯', label: t('nav.goals'), path: '/my-goals' },
    { icon: '✦',  label: t('menu_kaal'), path: '/kaal' },
    { icon: '⚖️', label: t('menu_niti'), path: '/niti' },
    { icon: '⋯',  label: t('more.label'), path: null,
      action: () => setMoreTrayOpen(true) },
  ];

  return (
    <>
      <nav className="bottom-nav">
        {navItems.map(item => (
          <button
            key={item.label}
            className={`bn-item ${
              item.path && location.pathname === item.path
                ? 'active' : ''
            }`}
            onClick={() => {
              if (item.action) item.action();
              else navigate(item.path!);
            }}
          >
            <span className="bn-icon">{item.icon}</span>
            <span className="bn-label">{item.label}</span>
          </button>
        ))}
      </nav>

      <MoreTray
        isOpen={moreTrayOpen}
        onClose={() => setMoreTrayOpen(false)}
      />
    </>
  );
}
```
