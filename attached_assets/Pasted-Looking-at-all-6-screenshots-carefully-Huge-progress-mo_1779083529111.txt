## The Mobile Layout Fix — Most Important

This is the biggest UX issue. Here's the fix:

```tsx
// Make all cards collapsible on mobile
// Chat input always visible and full-width

function HomeScreen() {
  const [showCheckin, setShowCheckin] = useState(true);
  const [showSuggestions, setShowSuggestions] = useState(false);

  return (
    <div className="home-screen">

      {/* Collapsible Daily Check-in */}
      <div className="collapsible-section">
        <button
          className="section-toggle"
          onClick={() => setShowCheckin(!showCheckin)}
        >
          😊 {t('mood.checkin')}
          <span>{showCheckin ? '▲' : '▼'}</span>
        </button>
        {showCheckin && <MoodCheckInCard />}
      </div>

      {/* Collapsible Suggestions */}
      <div className="collapsible-section">
        <button
          className="section-toggle"
          onClick={() => setShowSuggestions(!showSuggestions)}
        >
          💡 {t('suggest.label')}
          <span>{showSuggestions ? '▲' : '▼'}</span>
        </button>
        {showSuggestions && <SuggestionCards />}
      </div>

      {/* Chat — always visible, takes remaining space */}
      <div className="chat-area">
        <ChatMessages />
      </div>

      {/* Input — always at bottom, comfortable height */}
      <div className="chat-input-bar">
        <textarea
          className="chat-input"
          placeholder={t('chat.placeholder')}
          rows={2}           // minimum 2 rows
          style={{
            minHeight: '52px',
            maxHeight: '120px',
            fontSize: '16px'  // prevents iOS zoom
          }}
        />
      </div>

    </div>
  );
}
```

```css
/* Mobile chat input — comfortable size */
@media (max-width: 768px) {
  .chat-input-bar {
    padding: 12px 16px;
    padding-bottom: env(safe-area-inset-bottom, 16px);
    /* accounts for iPhone home bar */
  }

  .chat-input {
    font-size: 16px;  /* critical — prevents iOS auto-zoom */
    min-height: 52px;
    padding: 14px 16px;
    border-radius: 16px;
    width: 100%;
  }

  /* Make suggestion cards horizontal scroll
     instead of vertical stack */
  .suggestions-grid {
    display: flex;
    overflow-x: auto;
    gap: 10px;
    padding-bottom: 8px;
    scrollbar-width: none;
  }
  .suggestion-card {
    min-width: 200px;
    flex-shrink: 0;
  }
}
```

    icon: '💎'

function NitiCategory({ category, lang }) {
  return (
    <div className="niti-category">
      <span className="cat-icon">{category.icon}</span>
      <div className="cat-text">
        {lang === 'hi' ? (
          <>
            <span className="cat-main">{category.hi}</span>
            <span className="cat-sub">({category.en})</span>
          </>
        ) : (
          <span className="cat-main">{category.en}</span>
        )}
      </div>
    </div>
  );
}
