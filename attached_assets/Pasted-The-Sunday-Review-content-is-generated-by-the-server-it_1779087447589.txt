## Exact Fix — weekly-review.ts

Your developer needs to find where the Sunday review content is generated and add language awareness. Here is the exact change:

```typescript
// server/arya/weekly-review.ts

// STEP 1 — Get user's language before generating
async function generateWeeklyReview(userId: string) {

  // Get user data including language preference
  const user = await db.query(
    `SELECT * FROM arya_users WHERE id = $1`,
    [userId]
  );

  const userLanguage = user.ui_language || 'en';
  const userName = user.name || 'friend';

  // STEP 2 — Build language instruction
  const langInstruction = getWeeklyReviewLanguage(
    userLanguage,
    userName
  );

  // STEP 3 — Inject into EVERY section prompt
  // Opening reflection
  const openingPrompt = `
    ${langInstruction}
    Write a 1-2 line personal opening reflection
    for ${userName}'s weekly review.
    Based on their week: ${weekSummary}
  `;

  // KAAL cosmic section
  const kaalPrompt = `
    ${langInstruction}
    Write the cosmic week insight for ${userName}.
    Planet this week: ${planet}
    Keep planet names in English (Saturn, Jupiter etc).
    2-3 sentences. Slightly poetic but simple.
  `;

  // Pattern insight
  const patternPrompt = `
    ${langInstruction}
    Write ARYA's pattern observation for ${userName}.
    Data: ${patternData}
    Direct, caring, 2-3 sentences.
  `;

  // ARYA's question
  const questionPrompt = `
    ${langInstruction}
    Write one reflective question for ${userName}
    to sit with this week.
    Conversational, personal, 1 sentence.
  `;
}

// STEP 4 — The language instruction function
function getWeeklyReviewLanguage(
  langCode: string,
  userName: string
): string {

  const instructions: Record<string, string> = {

    hi: `
      LANGUAGE RULE — NON-NEGOTIABLE:
      Generate ALL content in Hindi.
      Tone: एक करीबी दोस्त की तरह — "तुम" use करो, "आप" नहीं।
      Natural Hinglish is fine — keep these in English:
        - ARYA, Niti, KAAL, Saturn, Jupiter, Mars, Venus
        - Pro, Goals, Business Mind
      Opening: थोड़ा काव्यात्मक लेकिन सरल
      Question: बिल्कुल conversational
      Pattern: सीधा और caring
      Example opening:
        "यार ${userName}, ये हफ्ता थोड़ा शांत था —
         लेकिन शांति भी कुछ कहती है।"
      Example question:
        "इस हफ्ते एक काम था जो तुम करना चाहते थे
         लेकिन नहीं हुआ — वो क्या था?"
    `,

    ta: `
      LANGUAGE RULE:
      Generate ALL content in Tamil.
      Tone: ஒரு நெருங்கிய நண்பனிடம் பேசுவது போல்.
      Keep: ARYA, Niti, KAAL, Saturn, Jupiter in English.
    `,

    ml: `
      LANGUAGE RULE:
      Generate ALL content in Malayalam.
      Tone: ഒരു അടുത്ത ചങ്ങാതിയോട് സംസാരിക്കുന്നതു പോലെ.
      Keep: ARYA, Niti, KAAL, Saturn, Jupiter in English.
    `,

    te: `
      LANGUAGE RULE:
      Generate ALL content in Telugu.
      Tone: ఒక దగ్గరి స్నేహితుడితో మాట్లాడినట్లు.
    `,

    kn: `
      LANGUAGE RULE:
      Generate ALL content in Kannada.
      Tone: ಆತ್ಮೀಯ ಸ್ನೇಹಿತನ ಹಾಗೆ.
    `,

    bn: `
      LANGUAGE RULE:
      Generate ALL content in Bengali.
      Tone: একজন কাছের বন্ধুর মতো।
    `,

    mr: `
      LANGUAGE RULE:
      Generate ALL content in Marathi.
      Tone: एक जवळच्या मित्रासारखे.
    `,

    // Global languages
    ar: `
      LANGUAGE RULE:
      Generate ALL content in Arabic.
      Use Gulf dialect — warm, conversational.
      Never formal fusha in casual sections.
      Keep: ARYA, Niti, KAAL in English.
    `,

    fr: `
      LANGUAGE RULE:
      Generate ALL content in French.
      Always use "tu" not "vous".
      Natural conversational French.
    `,

    es: `
      LANGUAGE RULE:
      Generate ALL content in Spanish.
      Use "tú" not "usted".
      Natural Latin American Spanish.
    `,

    de: `
      LANGUAGE RULE:
      Generate ALL content in German.
      Use "du" not "Sie".
      Warm, direct German.
    `,

    // Default
    en: `Generate all content in English.`
  };

  return instructions[langCode] || instructions['en'];
}
```
