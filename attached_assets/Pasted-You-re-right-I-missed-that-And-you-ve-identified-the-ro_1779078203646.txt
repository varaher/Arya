## Fix 4 — The Complete Language Switch Function

This is the unified fix. When ANY language is selected — Indian or global — everything changes.

```typescript
// client/src/lib/language-manager.ts

export async function applyLanguage(langCode: string): Promise<void> {

  const INDIAN_LANGUAGES = [
    'hi','ta','te','kn','ml','bn','mr','gu','pa','or','sa'
  ];
  const RTL_LANGUAGES = ['ar', 'he'];

  // 1. Switch i18n — triggers ALL UI string updates
  //    Works for all 25 languages if locale files exist
  await i18n.changeLanguage(langCode);

  // 2. Set document direction
  const isRTL = RTL_LANGUAGES.includes(langCode);
  document.documentElement.dir = isRTL ? 'rtl' : 'ltr';
  document.documentElement.lang = langCode;

  // 3. Load correct font
  await loadLanguageFont(langCode);

  // 4. Apply language-specific CSS class to body
  //    Allows CSS to target specific scripts
  document.body.className = document.body.className
    .replace(/lang-\w+/g, '')
    .trim();
  document.body.classList.add(`lang-${langCode}`);

  // 5. Update CSS variable for font
  const fontFamily = LANGUAGE_FONTS[langCode] || 'Manrope';
  document.documentElement.style
    .setProperty('--font-primary', `'${fontFamily}', sans-serif`);

  // 6. Save to localStorage
  localStorage.setItem('arya_ui_language', langCode);
  localStorage.setItem('arya_response_language', langCode);

  // 7. Save to DB — both UI and response language
  fetch('/api/user/preferences', {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      ui_language: langCode,
      response_language: langCode
    })
  }).catch(console.error);

  // 8. Notify chat component to update globe
  window.dispatchEvent(
    new CustomEvent('arya-language-change', {
      detail: { langCode }
    })
  );
}
```
