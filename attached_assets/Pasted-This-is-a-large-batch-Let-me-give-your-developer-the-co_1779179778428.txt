```tsx
      path: '/niti',
      match: (p: string) => p.startsWith('/niti')

      icon: '⋯',
      label: t('more_label'),
      path: null,
      match: () => false
    },
  ];

  return (
    <>
      <nav className="bottom-nav">
        {items.map((item, i) => {
          const isActive = item.path
            ? item.match(location)
            : moreTrayOpen;

          return (
            <button
              key={i}
              className={`bn-btn ${isActive ? 'bn-btn--active' : ''}`}
              onClick={() => {
                if (!item.path) {

                  navigate(item.path);

              <span className="bn-icon">{item.icon}</span>
              <span className="bn-label">{item.label}</span>
            </button>
          );
        })}
      </nav>

      <MoreTray
        isOpen={moreTrayOpen}
        onClose={() => setMoreTrayOpen(false)}
        onOpenCustomize={onOpenCustomize}
      />
    </>
  );
}
```

```tsx
// client/src/pages/admin/SelfLearningPage.tsx
// New file — complete admin UI

import { useState, useEffect } from 'react';

interface KnowledgeGap {
  id: string;
  trigger_query: string;
  failed_response: string;
  correction_hint?: string;
  auto_draft?: string;
  gap_category: string;
  frequency: number;
  quality_score_at_flag: number;
  status: string;
  created_at: string;
}

interface QualityReport {
  golden: number;
  neutral: number;
  flagged: number;
  total: number;
  cache_hit_rate: number;
}

export function SelfLearningPage() {
  const [gaps, setGaps] = useState<KnowledgeGap[]>([]);
  const [report, setReport] = useState<QualityReport | null>(null);
  const [activeGap, setActiveGap] = useState<KnowledgeGap | null>(null);
  const [editedDraft, setEditedDraft] = useState('');
  const [loading, setLoading] = useState(true);
  const [filter, setFilter] = useState<'all' | 'draft_ready' | 'detected'>('draft_ready');

  useEffect(() => {
    loadData();
  }, []);

  async function loadData() {
    setLoading(true);
    try {
      const [gapsRes, reportRes] = await Promise.all([
        fetch('/api/arya/knowledge-gaps'),
        fetch('/api/arya/cache/quality-report'),
      ]);
      setGaps(await gapsRes.json());
      setReport(await reportRes.json());
    } finally {
      setLoading(false);
    }
  }

  async function approveGap(gap: KnowledgeGap) {
    await fetch(`/api/arya/knowledge-gaps/${gap.id}/approve`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ approved_response: editedDraft || gap.auto_draft })
    });
    setActiveGap(null);
    loadData();

                ✓ Approve & Save

            </div>

          </div>

      )}

    </div>
  );
}
```

### Backend Endpoints For Self-Learning Panel

```typescript
// server/routes.ts — add these

// Get all knowledge gaps
app.get('/api/arya/knowledge-gaps',
  requireAdmin, async (req, res) => {
    const gaps = await db.query(`
      SELECT * FROM knowledge_gaps
      ORDER BY
        CASE status
          WHEN 'draft_ready' THEN 1
          WHEN 'detected' THEN 2
          ELSE 3
        END,
        frequency DESC,
        created_at DESC
      LIMIT 100
    `);
    res.json(gaps.rows);
  }
);

// Approve a gap
app.post('/api/arya/knowledge-gaps/:id/approve',
  requireAdmin, async (req, res) => {
    const { id } = req.params;
    const { approved_response } = req.body;

    await db.query(`
      UPDATE knowledge_gaps
      SET status = 'approved',
          auto_draft = $1,
          admin_approved = true,
          reviewed_at = now()
      WHERE id = $2
    `, [approved_response, id]);

    // Add to knowledge base for RAG
    await db.query(`
      INSERT INTO arya_knowledge
        (content, category, source, created_at)
      SELECT
        trigger_query || ' ' || $1,
        gap_category,
        'self_learning',
        now()
      FROM knowledge_gaps WHERE id = $2
    `, [approved_response, id]);

    res.json({ success: true });
  }
);

// Reject a gap
app.post('/api/arya/knowledge-gaps/:id/reject',
  requireAdmin, async (req, res) => {
    await db.query(`
      UPDATE knowledge_gaps
      SET status = 'rejected',
          admin_approved = false,
          reviewed_at = now()
      WHERE id = $1
    `, [req.params.id]);

    res.json({ success: true });
  }
);
```
