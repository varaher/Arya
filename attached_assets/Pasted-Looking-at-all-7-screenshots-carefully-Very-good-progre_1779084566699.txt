function QuickAccessTabs() {
  const { t } = useTranslation();
  const navigate = useNavigate();

    { icon: '🎯', label: t('nav.goals'),    path: '/goals' },
    { icon: '🧠', label: t('nav.memory'),   path: '/memory' },
    { icon: '📝', label: t('nav.notes'),    path: '/notes' },

    { icon: '🔔', label: t('nav.reminders'),path: '/reminders' },
    { icon: '📅', label: t('nav.calendar'), path: '/calendar' },
  ];

  return (
    <div className="quick-tabs">
      {tabs.map(tab => (
        <button
          key={tab.path}
          className="quick-tab-pill"
          onClick={() => navigate(tab.path)}
        >
          <span>{tab.icon}</span>
          <span>{tab.label}</span>
        </button>
      ))}
    </div>
  );
}

### Request 2 — Chat toolbar: make globe, attachment, headphones a dropdown

Keep mic and mute separate. Everything else collapses into one "⋯" button.

```tsx
// ChatToolbar.tsx — restructure

function ChatToolbar({ onSend, language, onLanguageChange }) {
  const [showMore, setShowMore] = useState(false);

  return (
    <div className="chat-toolbar">

      {/* Primary — always visible */}
      <button className="tool-mic" title="Voice input">
        🎙️
      </button>
      <button className="tool-mute" title="Mute">
        🔇
      </button>

      {/* More options — dropdown */}
      <div className="tool-more-wrap">
        <button
          className="tool-more-btn"
          onClick={() => setShowMore(!showMore)}
          title="More options"
        >
          ⋯
        </button>

        {showMore && (
          <div className="tool-more-dropdown">
            <button onClick={() => { openGlobe(); setShowMore(false); }}>
              🌐 {t('toolbar.language')}
            </button>
            <button onClick={() => { openAttachment(); setShowMore(false); }}>
              📎 {t('toolbar.attach')}
            </button>
            <button onClick={() => { openHeadphones(); setShowMore(false); }}>
              🎧 {t('toolbar.listen')}
            </button>
          </div>
        )}
      </div>

      {/* Input field */}
      <textarea
        className="chat-input"
        placeholder={t('chat.placeholder')}
      />

      {/* Send */}
      <button className="tool-send">→</button>

    </div>
  );
}
```
