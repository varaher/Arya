### 1. Response Quality Scoring — Highest Impact

**What it does:** Every ARYA response gets a quality score based on what happens after it. High-scoring responses become golden cache. Low-scoring ones get flagged for review.

```typescript
// response-quality-scorer.ts

interface QualitySignal {
  response_id: string;
  user_id: string;
  explicit_rating?: 'up' | 'down';      // thumbs up/down
  continued_conversation: boolean;        // did they reply?
  session_continued: boolean;            // did they keep using ARYA?
  bookmarked: boolean;                   // did they save it?
  returned_next_day: boolean;            // came back tomorrow?
  follow_up_within_30s: boolean;         // immediate follow-up = confused
}

function calculateQualityScore(signals: QualitySignal): number {
  let score = 50; // baseline

  // Explicit signals (strongest)
  if (signals.explicit_rating === 'up')   score += 30;
  if (signals.explicit_rating === 'down') score -= 40;

  // Implicit engagement signals
  if (signals.continued_conversation)     score += 10;
  if (signals.session_continued)          score += 8;
  if (signals.bookmarked)                 score += 15;
  if (signals.returned_next_day)          score += 12;

  // Negative signals
  if (signals.follow_up_within_30s)      score -= 8;
  // (immediate follow-up usually means confusion)

  return Math.max(0, Math.min(100, score));
}

// Threshold: score > 75 → promote to golden cache
// score < 30 → flag for admin review in Self-Learning panel
```

**UI change:** One thumbs up / thumbs down under every ARYA response. Already designed in your admin Self-Learning panel. Just needs wiring to this scorer.

### 2. Auto Language Detection — Already Discussed

```typescript
// language-detector.ts

const SCRIPT_PATTERNS = {
  hi: /[\u0900-\u097F]/,    // Devanagari (Hindi, Marathi, Sanskrit)
  ta: /[\u0B80-\u0BFF]/,    // Tamil
  te: /[\u0C00-\u0C7F]/,    // Telugu
  kn: /[\u0C80-\u0CFF]/,    // Kannada
  ml: /[\u0D00-\u0D7F]/,    // Malayalam
  bn: /[\u0980-\u09FF]/,    // Bengali
  gu: /[\u0A80-\u0AFF]/,    // Gujarati
  pa: /[\u0A00-\u0A7F]/,    // Gurmukhi (Punjabi)
  or: /[\u0B00-\u0B7F]/,    // Odia
  ar: /[\u0600-\u06FF]/,    // Arabic
  he: /[\u0590-\u05FF]/,    // Hebrew
  ja: /[\u3040-\u30FF\u4E00-\u9FAF]/, // Japanese
  zh: /[\u4E00-\u9FFF]/,    // Chinese
  ko: /[\uAC00-\uD7AF]/,    // Korean
  ru: /[\u0400-\u04FF]/,    // Cyrillic (Russian)
};

function detectLanguage(text: string): string {
  for (const [lang, pattern] of Object.entries(SCRIPT_PATTERNS)) {
    if (pattern.test(text)) return lang;
  }
  return 'en'; // default
}

// Per-user language profile in DB
interface UserLanguageProfile {
  user_id: string;
  ui_language: string;          // what they set in settings
  detected_languages: {         // what they actually write in
    [lang: string]: number;     // count of messages per language
  };
  preferred_response_language: string; // derived: most used detected
  last_detected: string;
  auto_switch_enabled: boolean; // user preference
}

// Update on every message
async function updateLanguageProfile(
  userId: string,
  messageText: string
) {
  const detected = detectLanguage(messageText);

  await db.userLanguageProfiles.upsert({
    where: { user_id: userId },
    update: {
      [`detected_languages.${detected}`]: { increment: 1 },
      last_detected: detected
    }
  });

  // After 3+ messages in same language → auto-switch response language
  const profile = await getProfile(userId);
  const count = profile.detected_languages[detected] || 0;

  if (count >= 3 && profile.auto_switch_enabled) {
    await updatePreferredResponseLanguage(userId, detected);
  }
}
```

**What the user experiences:** They start writing in Tamil even though their UI is in English. After 3 messages ARYA smoothly starts responding in Tamil. No settings change required. It just happens.

### 3. Section-Level Personalisation — Per Feature Learning

Each section of ARYA learns independently what works for this specific user.

```typescript
// section-personaliser.ts

// GOALS SECTION
interface GoalPersonalisation {
  user_id: string;
  // What goal types this user actually completes
  successful_goal_types: string[];    // ['health', 'reading', 'exercise']
  failed_goal_types: string[];        // ['social', 'work-tasks']
  optimal_goal_count: number;         // 3 goals → 89% completion vs 6 → 12%
  best_check_in_time: string;        // '08:00' — when they actually log
  streak_motivation_style: string;   // 'milestone' | 'streak' | 'quiet'
}

// MOOD SECTION
interface MoodPersonalisation {
  user_id: string;
  avg_mood_by_day: { [day: string]: number };   // Mon: 3.2, Fri: 4.1
  avg_mood_by_hour: { [hour: string]: number }; // 8am: 4.2, 10pm: 2.8
  mood_triggers_positive: string[];  // from memory engine
  mood_triggers_negative: string[];
  check_in_completion_rate: number;  // 0.73 = checks in 73% of days
}

// BRIEFING SECTION
interface BriefingPersonalisation {
  user_id: string;
  // Which briefing sections they actually read
  engagement_by_section: {
    goals_recap: number;      // 0.9 = reads 90% of the time
    kaal_timing: number;      // 0.4 = skips often
    mood_trend: number;       // 0.8
    news_cards: number;       // 0.3 = mostly ignores
  };
  // Which news topics they tap on
  engaged_news_categories: string[];
  briefing_open_rate: number;
  avg_time_spent_seconds: number;
}

// How ARYA uses this:
function buildPersonalisedPrompt(
  user: User,
  section: string,
  personalisation: any
): string {
  if (section === 'goals') {
    return `
This user completes ${personalisation.successful_goal_types.join(', ')} goals
reliably. They struggle with ${personalisation.failed_goal_types.join(', ')}.
Optimal goal count for them: ${personalisation.optimal_goal_count}.
Suggest goals within their successful types. Don't push more than
${personalisation.optimal_goal_count} active goals.
    `;
  }

  if (section === 'briefing') {
    const lowEngagement = Object.entries(
      personalisation.engagement_by_section
    )
    .filter(([_, score]) => (score as number) < 0.3)
    .map(([section]) => section);

    return `
Deprioritise these briefing sections (user rarely reads them):
${lowEngagement.join(', ')}.
Lead with what they actually engage with.
    `;
  }

  return '';
}
```

### 4. Knowledge Gap Auto-Fill — Connecting learning-engine.ts

Your `learning-engine.ts` already detects when ARYA struggles. This closes the loop.

```typescript
// knowledge-gap-handler.ts

interface KnowledgeGap {
  id: string;
  trigger_query: string;       // what the user asked
  arya_response_quality: number; // scored by quality engine
  gap_category: string;        // 'indian_startups' | 'health' | 'astrology'
  frequency: number;           // how many users hit this gap
  status: 'detected' | 'draft_ready' | 'approved' | 'live';
  auto_draft?: string;         // generated by Claude to fill the gap
  admin_approved: boolean;
}

// When quality score < 35 on a response:
async function handleLowQualityResponse(
  responseId: string,
  userQuery: string,
  aryaResponse: string
) {
  // 1. Flag the gap
  const gap = await db.knowledgeGaps.create({
    trigger_query: userQuery,
    arya_response_quality: await getScore(responseId),
    gap_category: await categoriseQuery(userQuery),
    frequency: 1,
    status: 'detected'
  });

  // 2. Auto-generate a better answer using Claude
  const draft = await generateDraft(userQuery, gap.gap_category);

  // 3. Send to admin Self-Learning panel for approval
  await db.knowledgeGaps.update({
    where: { id: gap.id },
    data: {
      auto_draft: draft,
      status: 'draft_ready'
    }
  });

  // 4. Admin approves → goes into knowledge base → future queries
  //    get cached golden answer instead of calling OpenAI
}

// Cost impact:
// First user asks → OpenAI call → low quality → gap detected → draft
// Admin approves → cached
// Next 1000 users ask same thing → zero API calls → golden answer
// This is your real "ML" — it just runs on Claude + PostgreSQL
```

## How All Four Connect

```
User sends message
        ↓
Language detector runs → updates language profile
        ↓
Section personaliser pulls user's behavioural data
        ↓
System prompt assembled with:
  - Base ARYA instructions
  - Language + tone instruction
  - Section personalisation context
  - Relevant memories from memory-engine
  - User's pattern data from patterns-engine
        ↓
Check response cache first
  → Cache hit (score > 75): return golden answer instantly
  → Cache miss: call Claude API
        ↓
Response delivered to user
        ↓
Quality signals collected (implicit + explicit)
        ↓
Quality scorer runs → updates response score
        ↓
Score > 75: promote to golden cache
Score < 35: flag as knowledge gap → auto-draft → admin review
        ↓
Cycle repeats → ARYA gets measurably better
```
